		{

				if(heap.size() < k)

					heap.pop(),heap.push(x);
			}
			else

	}

#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<queue>
using namespace std;
typedef long long LL;
int n,n1,n2;

int main()
{
	while(~scanf("%d%d%d",&n1,&n2,&n))
	{
		if(!n1 && !n2 && !n) break;
		
		priority_queue<int,vector<int>,greater<int> > heap1;
		priority_queue<int> heap2;
		
		LL sum=0;
		for(int i=0;i<n;i++)
		{
			int x;
			scanf("%d",&x);
			sum+=x;
			
			if(heap1.size() < n1) heap1.push(x);
			else if(heap1.top() < x)
				heap1.pop(),heap1.push(x);
			
			if(heap2.size() < n2) heap2.push(x);
			else if(heap2.top() > x)
				heap2.pop(),heap2.push(x);
		}
		while(!heap1.empty()) sum-=heap1.top(),heap1.pop();
		while(!heap2.empty()) sum-=heap2.top(),heap2.pop();
		
		printf("%.6f\n",(double)sum/(n-n1-n2));
	}
}

#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<queue>
using namespace std;
const int N=10010;
typedef pair<int,int> PII;
PII sta[N];
int n,L,P;

int main()
{
	cin>>n;
	
	for(int i=0;i<n;i++)
	cin>>sta[i].first>>sta[i].second;

	cin>>L>>P;
	
	sort(sta,sta+n,greater<PII>());
	n++;
	sta[n-1]={0,0};
	
	int res=0;
	int oil=P;
	int last=L;
	priority_queue<int> heap;
	bool flag=true;
	for(int i=0;i<n;i++)
	{
		oil-=last-sta[i].first;	
		//cout<<last-sta[i].first<<endl;
		last=sta[i].first;
	
		while(oil < 0)
		{
			if(heap.empty())
			{
				flag=false;
				break;
			}
			int t=heap.top();
			heap.pop();
			res++;
			oil+=t;
		}
		
		if(!flag) break;
		
		heap.push(sta[i].second);
	}
	if(!flag) res=-1;
	printf("%d\n",res);

	return 0;
}

把每头奶牛按照minSPF[i]的值从小到大排序，把防晒霜也按照SPF[i]的值从小到大排序，加一个优先队列，让maxSPF[i]小的奶牛先使用防晒霜，因为maxSPF大的奶牛有更大的选择空间。用一个最小堆q维护maxSPF的最小值，可以高效解决问题。

#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<queue>
using namespace std;
const int N=2510;
typedef pair<int,int> PII;
PII cow[N];
PII bottle[N];
int n,m;

int main()
{
	cin>>n>>m;
	
	for(int i=0;i<n;i++) cin>>cow[i].first>>cow[i].second;
	
	for(int i=0;i<m;i++) cin>>bottle[i].first>>bottle[i].second;

	sort(cow,cow+n);
	sort(bottle,bottle+m);
	
	int res=0;
	priority_queue<int,vector<int>,greater<int> > heap;
	for(int i=0,j=0;i<m;i++)
	{
		while(j<n && cow[j].first <= bottle[i].first)
		{
			heap.push(cow[j].second);
			j++;
		}
		
		while(!heap.empty() && bottle[i].second)
		{
			int spf=heap.top();
			heap.pop();
			if(spf >= bottle[i].first)
			{
				res++;
				bottle[i].second--;
			}
		}
	}
	
	cout<<res<<endl;
	
	return 0;
}
