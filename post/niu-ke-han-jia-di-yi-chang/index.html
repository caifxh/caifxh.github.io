G
eli拿到了一个仅由小写字母组成的字符串。
她想截取一段连续子串，这个子串包含至少k个相同的某个字母。
她想知道，子串的长度最小值是多少？

可以单独处理每种字母找 k 个的长度（双指针），然后求最小值。

#include<iostream>
#include<cstring>
#include<algorithm>
#include<cstdio>
using namespace std;
const int N=200010;
char a[N];
int cnt[30];
int n,k;

int main()
{
	cin>>n>>k;
	scanf("%s",a);
	
	int ans=n+1;
	
	for(int i=0;i<26;i++)
	{
		int l=0,r=0;
		memset(cnt,0,sizeof cnt);
		while(1)
		{
			while(r<n && cnt[i] < k)
				cnt[a[r++]-'a']++;
	
			if(cnt[i] < k) break;
			
			ans=min(ans,r-l);
			
			cnt[a[l++]-'a']--;
		}
	}
	if(ans > n) ans=-1;
	printf("%d\n",ans);
	
	return 0;
}

H

有两种情况，一个是最长0子串，一个是最长1子串，只要取其中最大值即可
双指针：

#include<iostream>
#include<cstring>
#include<algorithm>
#include<cstdio>
using namespace std;
const int N=200010;
char s[N];
int n,k;

int main()
{
	cin>>n>>k;
	
	scanf("%s",s);
	
	int sum0=0;
	int l=0,r=0;
	int ans=0;
	
	while(r<n)
	{
		while(r<n && sum0 <= k)
		{
			if(s[r] - '0' == 0)
			{
				if(sum0 == k) break;
				else sum0++;
			}
			r++;
		}
		
		ans=max(ans,r-l);
		
		if(s[l++]-'0' == 0) sum0--;
	}
	
	l=0,r=0;
	int sum1=0;
	
	while(r<n)
	{
		while(r<n && sum1 <= k)
		{
			if(s[r] - '0' == 1)
			{
				if(sum1 == k) break;
				else sum1++;
			}
			r++;
		}
		
		ans=max(ans,r-l);
		
		if(s[l++] -'0' == 1) sum1--;
	}
	
	printf("%d\n",ans);
	
	return 0;
}

题解
可以把面积为 1 的“好三角形”分为两类分开统计：两条边和两个坐标轴平行；只有一条边和某个坐标轴平行。

对于第二种情况，可以分别统计底边为 2 、高为 1 和底边为 1 、高为 2 的情况。要注意底边靠近边界时的特殊讨论。

#include<iostream>
#include<cstring>
#include<cstdio>
#include<algorithm>
using namespace std;
typedef long long LL;
const int mod=1e9+7;

int main()
{
	LL n,m;
	cin>>n>>m;
	
	LL ans=4*((n-2)*(m-1)%mod + (m-2)*(n-1)%mod)%mod;
	
	ans=(ans+2*(n-2)*(n-2)%mod*(m-1)%mod+2*(m-2)*(m-2)%mod*(n-1)%mod)%mod;
	
	ans=(ans+2*(n-1)*(n-2)%mod*(m-2)%mod+2*(m-1)*(m-2)%mod*(n-2)%mod)%mod;
	
	cout<<ans<<endl;
	
	return 0;
}

#include<iostream>
#include<cstring>
#include<cstdio>
#include<algorithm>
using namespace std;
const int N=1e5+10;
typedef long long LL;
vector<int> g[N];
int p[N],cnt[N];
int son[N];//子节点所处连通块中白点数目 
LL sum[N];
char s[N];
int n;
LL ans;

int find(int x)
{
	if(x != p[x])
		p[x]=find(p[x]);
	return p[x];
}

void merge(int a,int b)
{
	int pa=find(a);
	int pb=find(b);
	if(pa != pb)
	{
		p[pa]=pb;
		cnt[pb]+=cnt[pa];
	}
}

int main()
{
	cin>>n>>s+1;
	
	for(int i=1;i<=n;i++)
		p[i]=i,cnt[i]=1;
		
	for(int i=1;i<n;i++)
	{
		int a,b;
		cin>>a>>b;
		g[a].push_back(b);
		g[b].push_back(a);
		if(s[a] == 'W' && s[b] =='W')
			merge(a,b);	
	}	
	
	for(int i=1;i<=n;i++)
		if(s[i] == 'B')
		{
			int tot=0;
			for(int j=0;j<g[i].size();j++)
				if(s[g[i][j]] == 'W')
					son[++tot]=cnt[find(g[i][j])];
			
			for(int j=1;j<=tot;j++)
				sum[j]=sum[j-1]+son[j];
			
			ans+=sum[tot];//黑点为端点 
			for(int j=1;j<=tot;j++)
				ans+=son[j]*sum[j-1];//黑点为中间点 
		}
		
	printf("%lld\n",ans);
	
	return 0;
}

树形dp解法

dp[i][0]表示i的子孙节点到i路径上无黑点的个数，dp[i][1]表示i的子孙节点到i路径上有一个黑点的个数。

#include<iostream>
#include<cstring>
#include<cstdio>
#include<algorithm>
using namespace std;
typedef long long LL;
const int N=1e5+10;
LL f[N][2];
LL ans;
vector<int> g[N];
char s[N];
int n;

void dfs(int u,int fa)
{
	if(s[u] == 'W') f[u][0]=1,f[u][1]=0;
	else f[u][0]=0,f[u][1]=1;
	
	for(int i=0;i<g[u].size();i++)
	{
		int j=g[u][i];
		if(j == fa) continue;
		dfs(j,u);
		ans+=f[u][0]*f[j][1]+f[u][1]*f[j][0];
		if(s[u] == 'W') f[u][0]+=f[j][0],f[u][1]+=f[j][1];
		else f[u][0]=0,f[u][1]+=f[j][0];
	}
}

int main()
{
	cin>>n>>s+1;
	for(int i=1;i<n;i++)
	{
		int x,y;
		cin>>x>>y;
		g[x].push_back(y);
		g[y].push_back(x);
	}
	
	dfs(1,-1);
	
	printf("%lld\n",ans);
	
	return 0;
}

#include<iostream>
#include<string>
using namespace std;
const int N=300010;
typedef long long LL;
LL f[N];
int n,a,b,c;

int main()
{
	cin>>n>>a>>b>>c;
	
	string s;
	cin>>s;
	
	for(int i=0;i<n;i++)
	{
		if(i) f[i]=f[i-1];
		if(i >= 3 && s.substr(i-3,4) == "nico")
			f[i]=max(f[i],f[i-3]+a);
		if(i >= 5 && s.substr(i-5,6) == "niconi")
			f[i]=max(f[i],f[i-5]+b);
		if(i >= 9 && s.substr(i-9,10) == "niconiconi")
			f[i]=max(f[i],f[i-9]+c); 
	}
	
	cout<<f[n-1]<<endl;
	
	return 0;
}
