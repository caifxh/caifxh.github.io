#include<iostream>
#include<cstdio>
using namespace std;
const int N=55,mod=1e9+7;
char g[N][N];
int f[N][N];
int n,m;

int main()
{
	cin>>n>>m;
	
	for(int i=0;i<n;i++)
		for(int j=0;j<m;j++)
		{
			char c;
			cin>>c;
			g[i][j]=c;
		}
		
	f[0][0]=1;
		
	for(int i=0;i<n;i++)
		for(int j=0;j<m;j++)
		{
			if(g[i][j] == 'R') f[i][j+1]=(f[i][j+1]+f[i][j])%mod;
			if(g[i][j] == 'D') f[i+1][j]=(f[i+1][j]+f[i][j])%mod;
			if(g[i][j] == 'B') f[i+1][j]=(f[i+1][j]+f[i][j])%mod,f[i][j+1]=(f[i][j+1]+f[i][j])%mod;
		}
		
	cout<<f[n-1][m-1]<<endl;
	
	return 0;
}

#include<iostream>
#include<cstdio>
using namespace std;
const int N=100010;
bool st[N];
int cnt[N],f[N];
int n,m;

void init(int n)
{
	for(int i=2;i<=n;i++)
	{
		if(!st[i])
		{
			for(int j=i+i;j<=n;j+=i)
				st[j]=true;
		}
		else 
		{
			for(int j=i;j<=n;j+=i)
				cnt[j]++;
		}
		f[cnt[i]]++;
	}
}

int main()
{
	cin>>n>>m;
	
	init(n);
	
	while(m--)
	{
		int k;
		scanf("%d",&k);
		printf("%d\n",f[k]);
	}
	return 0;
}

题意：
给定长度为n的01串，任意两个1之间能够产生的能量为它们的下标差，求01串能量总和。

#include<iostream>
#include<cstdio>
using namespace std;
const int N=100010,mod=1e9+7;
typedef long long LL;
char str[N];
int pos[N];
int d[N];
int n;
 
int main()
{
    cin>>n;
    cin>>str+1;
     
    int cnt=0;
    for(int i=1;i<=n;i++)
        if(str[i] == '1')
            pos[cnt++]=i;
             
    for(int i=1;i<cnt;i++)
        d[i]=pos[i]-pos[i-1];
     
    int ans=0;
    for(int i=1;i<cnt;i++)
        ans=(ans+(LL)i*(cnt-i)%mod*d[i]%mod)%mod;
         
    cout<<ans<<endl;
     
    return 0;
}

#include<iostream>
#include<cstdio>
using namespace std;
const int N=100010,mod=1e9+7;
typedef long long LL;
char str[N];
int p[N];
int s[N];
int n;

int main()
{
	cin>>n;
	cin>>str+1;
	
	int cnt=0;
	for(int i=1;i<=n;i++)
		if(str[i] == '1')
			p[++cnt]=i;
			
	for(int i=1;i<=cnt;i++)
		s[i]=(s[i-1]+p[i])%mod;
	
	int ans=0;
	for(int i=1;i<=cnt;i++)
		ans=(ans+(LL)(i-1)*p[i]-s[i-1]+mod)%mod;
	
	cout<<ans<<endl;
	return 0;
}
