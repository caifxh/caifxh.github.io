#include<iostream>
#include<cstring>
#include<cstdio>
#include<algorithm>
using namespace std;
const int N=2e6+10;
typedef long long LL;
int p[N],d[N];
LL s[N];
int q[N];
bool ans[N];
int n;

int main()
{
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
    {
        scanf("%d%d",&p[i],&d[i]);
        p[i+n]=p[i];
        d[i+n]=d[i];
    }

    for(int i=1;i<=n*2;i++) s[i]=s[i-1]+p[i]-d[i];

    int hh=0,tt=-1;
    for(int i=1;i<=n*2;i++)//顺时针求长度为n的窗口内前缀和的最小值
    {
        if(hh <= tt && i - q[hh] >= n) hh++;

        while(hh <= tt && s[q[tt]] >= s[i])
            tt--;
        q[++tt]=i;
        if(i >= n && s[q[hh]] >= s[i-n])
            ans[i-n+1]=true;
    }

    d[0]=d[n];
    for(int i=n*2;i>=1;i--) s[i]=s[i+1]+p[i]-d[i-1];

    hh=0,tt=-1;
    for(int i=n*2;i>=1;i--)//逆时针求长度为n的窗口内后缀和的最小值
    {
        if(hh <= tt && q[hh]-i >= n) hh++;

        while(hh <= tt && s[q[tt]] >= s[i])
            tt--;
        q[++tt]=i;
        if(i<=n+1 && s[q[hh]] >= s[i+n])
            ans[i-1]=true;
    }

    for(int i=1;i<=n;i++)
        if(ans[i]) puts("TAK");
        else puts("NIE");
    return 0;   
}

int n, m;
int w[N], q[N];
int f[N];

int main()
{
    scanf("%d%d", &n, &m);
    for (int i = 1; i <= n; i ++ ) scanf("%d", &w[i]);

    int hh = 0, tt = 0;
    q[0]=0;
    for (int i = 1; i <= n; i ++ )
    {
        if (i - q[hh] > m) hh ++ ;
        f[i] = f[q[hh]] + w[i];
        while (hh <= tt && f[q[tt]] >= f[i]) tt -- ;
        q[ ++ tt] = i;
    }

    int res = INF;
    for (int i = n - m + 1; i <= n; i ++ ) res = min(res, f[i]);
    printf("%d\n", res);

}

#include<iostream>
#include<cstring>
#include<algorithm>
#include<cstdio>
using namespace std;
const int N=50010,INF=0x3f3f3f3f;
int n,t;
int a[N];
int f[N];
int q[N];

bool check(int mid)
{
    int hh=0,tt=-1;
    q[++tt]=0;
    for(int i=1;i<=n;i++)
    {
        if(hh <= tt && i - q[hh] > mid+1) hh++;
        f[i]=f[q[hh]]+a[i];
        while(hh <= tt && f[q[tt]] >= f[i])
            tt--;
        q[++tt]=i;
    }
    
    int res=INF;
    for(int i=n-mid;i<=n;i++)
        res=min(res,f[i]);
    
    return res<=t;
}

int main()
{
    cin>>n>>t;
    for(int i=1;i<=n;i++)
        cin>>a[i];

    int l=0,r=n;
    while(l < r)
    {
        int mid=l+r>>1;
        if(check(mid))
            r=mid;
        else
            l=mid+1;
    }
    
    cout<<l<<endl;
    return 0;
}

先用滑动窗口+单调队列求出每一行长度为k的连续序列中的最大值、最小值，
存到rmin[i][j],rmax[i][j]两个数组里，
再用相同的方法求这两个数组中每一列长度为k的连续序列的最大值、最小值存到cmin[i],cmax[i]中。
遍历cmin,cmax得出答案。

#include<iostream>
#include<cstring>
#include<cstdio>
#include<algorithm>
using namespace std;
const int N=1010,INF=0x3f3f3f3f;
int n,m,k;
int a[N][N];
int rmax[N][N],rmin[N][N];
int cmax[N],cmin[N];
int tmp[N];
int q[N];

void get(int a[],int b[],int len,int type)//0:最小值，1：最大值 
{
    int hh=0,tt=-1;
    for(int i=1;i<=len;i++)
    {
        if(hh <= tt && i-q[hh] >=k) hh++;
        while(hh <= tt &&(type?a[q[tt]]<=a[i]:a[q[tt]]>=a[i]))
            tt--;
        q[++tt]=i;
        b[i]=a[q[hh]];
    }
}

int main()
{
    scanf("%d%d%d",&n,&m,&k);

    for(int i=1;i<=n;i++)
        for(int j=1;j<=m;j++)
            scanf("%d",&a[i][j]);

    for(int i=1;i<=n;i++)
    {
        get(a[i],rmin[i],m,0);
        get(a[i],rmax[i],m,1);
    }

    int ans=INF;
    for(int i=k;i<=m;i++)
    {
        for(int j=1;j<=n;j++) tmp[j]=rmin[j][i];
        get(tmp,cmin,n,0);

        for(int j=1;j<=n;j++) tmp[j]=rmax[j][i];
        get(tmp,cmax,n,1);

        for(int j=k;j<=n;j++)
            ans=min(ans,cmax[j]-cmin[j]);
    }

    printf("%d\n",ans);

    return 0;   
}
