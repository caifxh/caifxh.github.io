高精度加法

当进行加法运算的数字较大时，受存储范围限制，无法使用int或longlong型变量进行存储。
考虑到加法运算的本质，不妨将数字拆开来存储。
对于一个超过变量存储数据范围的数字，可以将这个数字拆开，拆成一位一位的，或者是几位几位的存储到一个数组中， 用一个数组去表示一个数字

采用用顺序存储数据方式，会有一个问题：最高位进位变得很困难。
最高位在num[0]，所以一旦有进位，需要将num[]数组中每一位都往后移，时间复杂度O(len)
·倒序存储·：num[]数组从0到len-1依次存储整数个位到最高位。则最高位进位时，只需要num[len++]=t，t为最高位数字

进位：
方法一：设一个变量t，记录当前进位
边计算边进位和逐位相加不进位最后再统一进位效果是一样的如下，先进行一遍逐位相加。
然后，从0开始遍历一遍ans.num[]，将本位大于9的部分进位给更高位。
最后判断是否最高位有进位，有则ans.len++

模板题

#include<iostream>
#include<cstring>
#include<algorithm>
#include<cstdio>
using namespace std;
const int N=510;
char s[N];
struct bignum
{
	int m[N],len;
	bignum()
	{
		memset(m,0,sizeof m);
		len=0;
	}
};

bignum add(bignum a,bignum b)
{
	bignum c;
	c.len=max(a.len , b.len);
	for(int i=0;i<c.len;i++) c.m[i]=a.m[i]+b.m[i];
	for(int i=0;i<c.len;i++)
		if(c.m[i] > 9)
		{
			c.m[i+1]+=1;
			c.m[i]-=10;
		}
	if(c.m[c.len]) c.len++;
	return c;
}

int main()
{
	scanf("%s",s);
	bignum a,b;
	int len=strlen(s);
	a.len=len;
	for(int i=0;i<len;i++)
		a.m[len-i-1]=s[i]-'0';
	
	scanf("%s",s);
	len=strlen(s);
	b.len=len;
	for(int i=0;i<len;i++)
		b.m[len-i-1]=s[i]-'0';
	bignum c=add(a,b);
	for(int i=c.len-1;i>=0;i--)
		printf("%d",c.m[i]);
	puts("");
	return 0;
}

高精度减法

  • 判断正负
  • 基本和加法一模一样，只不过从进位变成退位
  • 退位后更新结果长度
    模板题
#include<iostream>
#include<cstring>
#include<algorithm>
#include<cstdio>
using namespace std;
const int N=10100;
char s[N];
struct bignum
{
	int m[N],len;
	bignum()
	{
		memset(m,0,sizeof m);
		len=0;
	}
};

int judge(bignum a,bignum b)
{
	if(a.len > b.len) return 1;
	if(a.len < b.len) return -1;
	for(int i=a.len-1;i>=0;i--)
	{
		if(a.m[i] > b.m[i]) return 1;
		if(a.m[i] < b.m[i]) return -1;
	}
	return 0;
}

bignum Minus(bignum a,bignum b)
{
	if(judge(a,b) == -1)
	{
		printf("-");
		return Minus(b,a);
	}

	bignum c;
	c.len=a.len;
	for(int i=0;i<c.len;i++) c.m[i]=a.m[i]-b.m[i];
	for(int i=0;i<c.len;i++)
		if(c.m[i] < 0)
		{
			c.m[i+1]-=1;
			c.m[i]+=10;
		}
	while(c.len > 1 && !c.m[c.len-1]) c.len--;
	return c;
}

int main()
{
	scanf("%s",s);
	bignum a,b;
	int len=strlen(s);
	a.len=len;
	for(int i=0;i<len;i++)
		a.m[len-i-1]=s[i]-'0';
	
	scanf("%s",s);
	len=strlen(s);
	b.len=len;
	for(int i=0;i<len;i++)
		b.m[len-i-1]=s[i]-'0';
	bignum c=Minus(a,b);
	for(int i=c.len-1;i>=0;i--)
		printf("%d",c.m[i]);
	puts("");
	return 0;
}

高精度乘法

1.高精*低精

将加法运算改为乘法运算即可。
注意：

  • 进位的运算
  • 最高位进位的处理，可能不止进一位
    模板题
#include<iostream>
#include<cstring>
#include<algorithm>
#include<cstdio>
using namespace std;
const int N=100010;
char s[N];
struct bignum
{
	int m[N],len;
	bignum()
	{
		memset(m,0,sizeof m);
		len=0;
	}
};

bignum mul(bignum a,int b)
{
	bignum c;
	c.len=a.len;
	for(int i=0;i<c.len;i++) c.m[i]=a.m[i]*b;
	for(int i=0;i<c.len;i++)
	{
		c.m[i+1]+=c.m[i]/10;
		c.m[i]%=10;
		if(i == c.len-1 && c.m[i+1]) c.len++;
	}

	return c;
}

int main()
{
	scanf("%s",s);
	bignum a;
	int len=strlen(s);
	a.len=len;
	for(int i=0;i<len;i++)
		a.m[len-i-1]=s[i]-'0';
	
	int b;
	scanf("%d",&b);

	bignum c=mul(a,b);
	for(int i=c.len-1;i>=0;i--)
		printf("%d",c.m[i]);
	puts("");
	return 0;
}

2.高精*高精

乘法竖式中，a.num[i]和b.num[j]相乘的结果，放在了ans.num[i+j]的位置。
a,b都为正整数时，长度为a.len和b.len的数字相乘得到ans，则

模拟上例竖式运算过程即可
{模板题](https://www.luogu.com.cn/problem/P1303}

#include<iostream>
#include<cstring>
#include<algorithm>
#include<cstdio>
using namespace std;
const int N=4010;
char s[N];
struct bignum
{
	int m[N],len;
	bignum()
	{
		memset(m,0,sizeof m);
		len=0;
	}
};

bignum mul(bignum a,bignum b)
{
	bignum c;
	c.len=a.len+b.len;
	for(int i=0;i<a.len;i++)
		for(int j=0;j<b.len;j++)
			c.m[i+j]+=a.m[i]*b.m[j];
	for(int i=0;i<c.len;i++)
	{
		c.m[i+1]+=c.m[i]/10;
		c.m[i]%=10;
	}
	while(c.len > 1 && !c.m[c.len-1]) c.len--;
	return c;
}

int main()
{
	scanf("%s",s);
	bignum a,b;
	int len=strlen(s);
	a.len=len;
	for(int i=0;i<len;i++)
		a.m[len-i-1]=s[i]-'0';
	
	scanf("%s",s);
	b.len=len=strlen(s);
	for(int i=0;i<len;i++)
		b.m[len-i-1]=s[i]-'0';
	bignum c=mul(a,b);
	for(int i=c.len-1;i>=0;i--)
		printf("%d",c.m[i]);
	puts("");
	return 0;
}

高精度除法

1.高精/低精

除法是从高位到低位进行运算的。使用一个临时变量来记录余数，注意细节
模板题

#include<iostream>
#include<cstring>
#include<algorithm>
#include<cstdio>
using namespace std;
const int N=100010;
char s[N];
struct bignum
{
	int m[N],len;
	bignum()
	{
		memset(m,0,sizeof m);
		len=0;
	}
};
int t;

bignum Div(bignum a,int b)
{
	bignum c;
	c.len=a.len;
	t=0;
	for(int i=c.len-1;i>=0;i--)
	{
		t=t*10+a.m[i];
		c.m[i]=t/b;
		t%=b;
	}
	while(c.len > 1 && !c.m[c.len-1]) c.len--;
	return c;
}

int main()
{
	scanf("%s",s);
	bignum a;
	int len=strlen(s);
	a.len=len;
	for(int i=0;i<len;i++)
		a.m[len-i-1]=s[i]-'0';
	
	int b;
	scanf("%d",&b);
	bignum c=Div(a,b);
	for(int i=c.len-1;i>=0;i--)
		printf("%d",c.m[i]);
	puts("");
	printf("%d\n",t);

	return 0;
}
