#include<iostream>
using namespace std;

int main()
{
	int n;
	cin>>n;
	int res=0;
	for(int i=1;i*i<=n;i++)
	{
		int t=i*i;
		for(int j=1;j*j<=t;j++)
			if(t % j == 0)
			{
				res++;
				if(j != t/j) res++;
			}
	}
	
	cout<<res<<endl;
	
	return 0;
}

#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
const int N=3e5+10,INF=0x3f3f3f3f;
int f[N];
int a[N];
int n,k;

int main()
{
	cin>>n>>k;
	
	for(int i=1;i<=n;i++)
		scanf("%d",&a[i]);
	sort(a+1,a+n+1);
	
	f[0]=0;
	for(int i=1;i<k;i++)
		f[i]=INF;
		
	int val=INF;
	for(int i=k;i<=n;i++)
	{
		val=min(val,f[i-k]-a[i-k+1]);
		f[i]=val+a[i];
	}
	
	cout<<f[n]<<endl;
	
	return 0;
}

#include<iostream>
#include<cstring>
#include<cstdio>
#include<algorithm>
using namespace std;
#define lc u<<1
#define rc u<<1|1
typedef long long LL;
const int N=2e5+10,mod=1e9+7;
struct Node1
{
	int l,r;
	LL k;
}tr1[N<<2];
struct Node2
{
	int l,r;
	LL val;
}tr2[N<<2];
struct f
{
	LL k,val;
};
int k[N],b[N];
int n,m;

void pushup(int u)
{
	tr1[u].k=(LL)tr1[lc].k*tr1[rc].k%mod;
	tr2[u].val=((LL)tr2[lc].val*tr1[rc].k%mod+tr2[rc].val)%mod;
}

void build(int u,int l,int r)
{
	tr1[u].l=tr2[u].l=l,tr1[u].r=tr2[u].r=r;
	if(l == r)
	{
		tr1[u].k=k[l];
		tr2[u].val=b[l];
		return;
	}
	int mid=l+r>>1;
	build(lc,l,mid);
	build(rc,mid+1,r);
	pushup(u);
}

void modify(int u,int x,int k,int b)
{
	if(tr1[u].l == tr1[u].r)
	{
		tr1[u].k=k;
		tr2[u].val=b;
		return;
	}
	int mid=tr1[u].l+tr1[u].r>>1;
	if(x<=mid) modify(lc,x,k,b);
	else modify(rc,x,k,b);
	pushup(u);
}

f query(int u,int l,int r)
{
	if(l == tr1[u].l && tr1[u].r == r)
		return {tr1[u].k, tr2[u].val};

	int mid=tr1[u].l+tr1[u].r>>1;
	if(r <= mid) 
		return query(lc,l,r);
	else if(l > mid) 
		return query(rc,l,r);
	else 
	{
		f left=query(lc,l,mid);
		f right=query(rc,mid+1,r);
		return {left.k*right.k%mod,(left.val*right.k%mod+right.val)%mod};
	}
}

int main()
{
	cin>>n>>m;
	
	for(int i=1;i<=n;i++) scanf("%d",&k[i]);
	for(int i=1;i<=n;i++) scanf("%d",&b[i]);
		
	build(1,1,n);	
	
	while(m--)
	{
		int t;
		scanf("%d",&t);
		if(t == 1)
		{
			int i,k,b;
			scanf("%d%d%d",&i,&k,&b);
			modify(1,i,k,b);
		}
		else 
		{
			int l,r;
			scanf("%d%d",&l,&r);
			auto t=query(1,l,r);
			printf("%d\n",(t.k+t.val)%mod);
		}
	}
	
	return 0;
}

#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
typedef pair<int,int> PII;
const int N=2e5+10;
int a[N],b[N];
PII c[N];
int n;

bool cmp(PII x,PII y)
{
	return x.first > y.first;
}

int main()
{
	cin>>n;
	for(int i=1;i<=n;i++)
		scanf("%d",&a[i]);
	for(int i=1;i<=n;i++)
		scanf("%d",&b[i]);
	for(int i=1;i<=n;i++)
		c[i].first=a[i]+b[i],c[i].second=i;
	
	sort(c+1,c+n+1,cmp);
	
	for(int i=2;i<=n;i+=2)
		printf("%d ",c[i].second);
	puts("");
	
	for(int i=1;i<=n;i+=2)
		printf("%d ",c[i].second);
	puts("");
	
	return 0;
}
